Try asking about a topic!"; // Simple matching: check if key includes a known question for(const q in aiQA){ if(key.includes(q)) { response = aiQA[q]; break; } } appendMessage("STERLA AI", response); }, 400); } function appendMessage(sender, msg){ const div = document.createElement("div"); div.innerHTML = `${sender}: ${msg}`; div.style.margin="5px 0"; chatBox.appendChild(div); chatBox.scrollTop = chatBox.scrollHeight; }
